Algoritma Singlepath 

May 30, 2024 jofanny
algoritma code

Berikut adalah contoh implementasi Antrian dengan Algoritma Singlepath


#include <iostream>

#include <vector>

#include <cstdlib>

using namespace std;

#define N 5

#define M 1000

struct Graph {

    int V;

    vector<vector<int > > adj;

};

typedef struct {

    int item[N];

    int front;

    int rear;

    int count;

} Queue;

Graph createGraph() {

    Graph graph;

    graph.V = N;

    graph.adj.resize(N, vector<int>(N, M));

    return graph;

}

void initialize(int asal, int Beban[], int Rute[]) {

    for (int i = 0; i < N; i++) {

        if (i + 1 == asal)

            Beban[i] = 0;

        else

            Beban[i] = M;

        Rute[i] = 0;

    }

}

void display(int Beban[], int Rute[]) {

    cout << "Beban: ";

    for (int i = 0; i < N; i++) {

        if (Beban[i] >= M)

            cout << "M ";

        else

            cout << Beban[i] << " ";

    }

    cout << endl;

    cout << "Rute : ";

    for (int i = 0; i < N; i++) {

        cout << Rute[i] << " ";

    }

    cout << endl;

}

void inisialisasi_queue(Queue *q) {

    q->front = q->rear = q->count = 0;

}

int Kosong(Queue *q) {

    return q->count == 0;

}

int Penuh(Queue *q) {

    return q->count == N;

}

void Enqueue(Queue *q, int x) {

    if (Penuh(q)) {

        cout << "Queue Penuh !" << endl;

        exit(1);

    } else {

        q->item[q->rear] = x;

        q->rear = (q->rear + 1) % N;

        q->count++;

    }

}

int Dequeue(Queue *q) {

    int temp;

    if (Kosong(q)) {

        cout << "Queue kosong !" << endl;

        temp = -1;

    } else {

        temp = q->item[q->front];

        q->front = (q->front + 1) % N;

        q->count--;

    }

    return temp;

}

int main() {

    int asal, tujuan;

    cout << "Masukkan node asal : ";

    cin >> asal;

    cout << "Masukkan node tujuan : ";

    cin >> tujuan;

    Graph G = createGraph();

    int Beban[N], Rute[N];

    initialize(asal, Beban, Rute);

    cout << "Beban dan Rute awal\n";

    display(Beban, Rute);

    Queue q;

    inisialisasi_queue(&q);

    Enqueue(&q, 1);

    Enqueue(&q, 2);

    Enqueue(&q, 3);

    cout << "Elemen yang di-dequeue: " << Dequeue(&q) << endl;

    cout << "Elemen yang di-dequeue: " << Dequeue(&q) << endl;

    return 0;

}
